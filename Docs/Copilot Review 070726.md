✅ 1. Sitemap Generator Script (Python)
This script automatically scans your public/ folder, finds all .html files, and generates a correct sitemap.xml.

It is different from the static sitemap I gave you earlier — this one updates automatically whenever you add/remove pages.

Create:
generate_sitemap.py (place it at repo root, next to generate_favicons.py)

python
import os
from datetime import datetime

PUBLIC_DIR = "public"
OUTPUT_FILE = os.path.join(PUBLIC_DIR, "sitemap.xml")
BASE_URL = "https://topmoneytools.com"

def generate_sitemap():
    urls = []

    for root, dirs, files in os.walk(PUBLIC_DIR):
        for file in files:
            if file.endswith(".html"):
                rel_path = os.path.relpath(os.path.join(root, file), PUBLIC_DIR)
                url = f"{BASE_URL}/{rel_path.replace(os.sep, '/')}"
                urls.append(url)

    with open(OUTPUT_FILE, "w", encoding="utf-8") as f:
        f.write('<?xml version="1.0" encoding="UTF-8"?>\n')
        f.write('<urlset xmlns="http://www.sitemaps.org/schemas/sitemap/0.9">\n\n')

        for url in urls:
            f.write("  <url>\n")
            f.write(f"    <loc>{url}</loc>\n")
            f.write(f"    <lastmod>{datetime.utcnow().strftime('%Y-%m-%d')}</lastmod>\n")
            f.write("    <priority>0.8</priority>\n")
            f.write("  </url>\n\n")

        f.write("</urlset>\n")

    print(f"Sitemap generated: {OUTPUT_FILE}")

if __name__ == "__main__":
    generate_sitemap()

Then create: generate_sitemap_routes.py

python
import json
from datetime import datetime

BASE_URL = "https://topmoneytools.com"
ROUTES_FILE = "routes.json"
OUTPUT_FILE = "public/sitemap.xml"

def generate_sitemap():
    with open(ROUTES_FILE, "r") as f:
        routes = json.load(f)["routes"]

    with open(OUTPUT_FILE, "w", encoding="utf-8") as f:
        f.write('<?xml version="1.0" encoding="UTF-8"?>\n')
        f.write('<urlset xmlns="http://www.sitemaps.org/schemas/sitemap/0.9">\n\n')

        for route in routes:
            f.write("  <url>\n")
            f.write(f"    <loc>{BASE_URL}{route}</loc>\n")
            f.write(f"    <lastmod>{datetime.utcnow().strftime('%Y-%m-%d')}</lastmod>\n")
            f.write("    <priority>0.8</priority>\n")
            f.write("  </url>\n\n")

        f.write("</urlset>\n")

    print(f"Sitemap generated: {OUTPUT_FILE}")

if __name__ == "__main__":
    generate_sitemap()
This gives you a React‑accurate sitemap.
